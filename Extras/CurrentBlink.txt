enum NpcSpawnData = "Type" {
    Single {
        id: u16,
        pos: vector<f32>,
        yaw: f32,
        teamName: string?,
        isBearer: boolean,
        companyIndex: u8?,
		ownerUserId: f64,
    },
    Batch {
        list: struct {
            id: u16,
            pos: vector<f32>,
            yaw: f32,
            teamName: string?,
            isBearer: boolean,
            companyIndex: u8?,
			ownerUserId: f64,
        }[]
    }
}

event NpcBatch {
    From: Server,
    Type: Unreliable,
    Call: ManyAsync,
    Data: (f64, struct {
        id: u16,
        pos: vector<f32>,
        yaw: f32,
    }[])
}
