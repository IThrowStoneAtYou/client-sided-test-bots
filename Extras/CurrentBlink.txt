enum NpcSpawnData = "Type" {
    Single {
        id: u16,
        pos: vector<f32>,
        yaw: f32
    },
    Batch {
        list: struct {
            id: u16,
            pos: vector<f32>,
            yaw: f32
        }[]
    }
}

event NpcBatch {
    From: Server,
    Type: Unreliable,
    Call: ManyAsync,

    Data: (f64, struct {
        id: u16,
        pos: vector<f32>,
        yaw: f32,

    }[])
}

-- Reliable spawn event (server → clients)
event NpcSpawn {
    From: Server,
    Type: Reliable,
    Call: ManyAsync,
    Data: NpcSpawnData
}

-- Reliable despawn event (server → clients)
event NpcDespawn {
    From: Server,
    Type: Reliable,
    Call: ManyAsync,
    Data: u16   -- shortId
}

event NpcOrderCommand {
    From: Client,
    Type: Reliable,
    Call: ManyAsync,
    Data: (vector, vector, u8, u8)
}

-- Server → Client: regiment count (single value)
event NpcRegimentCount {
    From: Server,
    Type: Reliable,
    Call: ManyAsync,
    Data: u16
}

event NpcReady {
    From: Client,
