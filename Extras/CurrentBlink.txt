event Test {
    From: Client,
    Type: Reliable,
    Call: SingleSync,
    Data: u8
}

enum NpcSpawnData = "Type" {
    Single {
        id: u16,
        pos: vector<f32>,
        yaw: f32
    },
    Batch {
        list: struct {
            id: u16,
            pos: vector<f32>,
            yaw: f32
        }[]
    }
}

event NpcBatch {
    From: Server,
    Type: Unreliable,
    Call: ManyAsync,
    Data: (f64, struct {
        id: u16,
        pos: vector<f32>,
        yaw: f32,
    }[])
}

event NpcSpawn {
    From: Server,
    Type: Reliable,
    Call: ManyAsync,
    Data: NpcSpawnData
}

event NpcDespawn {
    From: Server,
    Type: Reliable,
    Call: ManyAsync,
    Data: u16
}

-- companyIndex added as 5th value
event NpcOrderCommand {
    From: Client,
    Type: Reliable,
    Call: ManyAsync,
    Data: (vector, vector, u8, u8, u8)
}

-- Replaces NpcRegimentCount: full company list for this player
event NpcCompanySync {
    From: Server,
    Type: Reliable,
    Call: ManyAsync,
    Data: struct {
        index: u8,
        count: u16,
        slots: u16[]
    }[]
}

event NpcReady {
    From: Client,
    Type: Reliable,
    Call: ManyAsync
}
